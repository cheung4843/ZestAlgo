    def collect_anagrams(words, target):
    def is_anagram(x, y):
        return sorted(x) == sorted(y)

    return [word for word in words if is_anagram(word, target)]

    def fibonacci(n):
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci(n - 1) + fibonacci(n - 2)
