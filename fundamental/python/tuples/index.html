    for i in range(n):

                is_edge = False
                for d in dirs:
                    x, y = i + d[0], j + d[1]

                        is_edge = True
                        break
                if is_edge:
                    print('0 ', end='')
                else:
                    print('_ ', end='')
        print()

    if k != N - 1:
        print()
