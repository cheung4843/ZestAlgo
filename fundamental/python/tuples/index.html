dirs = ((0, 1), (0, -1), (1, 0), (-1, 0))

for k in range(N):
    n, m = map(int, input().split())
    pic = []
    for _ in range(n):
        pic.append(input().split())

    for i in range(n):
        for j in range(m):
            if pic[i][j] == '0':
                print('_ ', end='')
            else:
                is_edge = False
                for d in dirs:
                    x, y = i + d[0], j + d[1]
                    if 0 <= x < n and 0 <= y < m and pic[x][y] == '0':
                        is_edge = True
                        break
                if is_edge:
                    print('0 ', end='')
                else:
                    print('_ ', end='')
        print()

    if k != N - 1:
        print()
